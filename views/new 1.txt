class User(){
	private $id;
	private $nom;
	private $prenom;
	
	public static setId($varId){
		$this->id=$varId;
	}
	
	public static getId(){
		return $id;
	}
	
	public static setNom($varNom){
		$this->nom=$varNom;
	}
	
	public static getNom(){
		return $nom;
	}
	
	public static setPrenom($varPrenom){
		$this->prenom=$varPrenom;
	}
	
	public static getPrenom(){
		return $prenom;
	}

}
